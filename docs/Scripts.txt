CREATE DATABASE LABORATORIO;

CREATE TABLE PESQUISADOR(
	CPF 		VARCHAR(11) PRIMARY KEY NOT NULL,
	NOME		VARCHAR(40) NOT NULL,
	TITULACAO	VARCHAR(50),
	ENDERECO	VARCHAR(150),
	TELEFONE	VARCHAR(15),
	EMAIL		VARCHAR(30),
	BIOGRAFIA	VARCHAR(100),
	IS_ALUNO	TINYINT(1) NOT NULL,
	IS_PROFESSOR	TINYINT(1) NOT NULL,
	IS_COLABORADOR	TINYINT(1) NOT NULL
)
ENGINE = MyISAM;

CREATE TABLE ALUNO(
	CPF 	VARCHAR(11) PRIMARY KEY NOT NULL REFERENCES PESQUISADOR(CPF),
	RA		VARCHAR(10) NOT NULL
)
ENGINE = MyISAM

CREATE TABLE GRADUANDOIC(
	CPF 		VARCHAR(11) PRIMARY KEY NOT NULL REFERENCES ALUNO(CPF),
	DATA_INICIAL	DATE NOT NULL,
	DATA_FINAL	DATE
)
ENGINE = MyISAM;

CREATE TABLE MESTRANDO(
	CPF 		VARCHAR(11) PRIMARY KEY NOT NULL REFERENCES ALUNO(CPF),
	DATA_INICIAL	DATE NOT NULL,
	DATA_FINAL	DATE
)
ENGINE = MyISAM;

CREATE TABLE DOUTORANDO(
	CPF 		VARCHAR(11) PRIMARY KEY NOT NULL REFERENCES ALUNO(CPF),
	DATA_INICIAL	DATE NOT NULL,
	DATA_FINAL	DATE
)
ENGINE = MyISAM;

CREATE TABLE PROFESSOR(
	CPF 				VARCHAR(11) PRIMARY KEY NOT NULL,
	DATA_INICIAL			DATE  NOT NULL,
	DATA_FINAL			DATE,
	NOME_LABORATORIO		VARCHAR(30),
	DATA_INICIAL_COORDENA_LAB	DATE,
	DATA_FINAL_COORDENA_LAB		DATE,
	OWNER VARCHAR(11) NOT NULL REFERENCES PESQUISADOR(CPF)
)
ENGINE = MyISAM;

CREATE TABLE LABORATORIO(
	NOME 		VARCHAR(30) PRIMARY KEY NOT NULL,
	ENDERECO	VARCHAR(150)
)
ENGINE = MyISAM;

CREATE TABLE COLABORADOR(
	CPF 			VARCHAR(11) PRIMARY KEY NOT NULL REFERENCES PESQUISADOR(CPF),
	IS_POS_DOUTORANDO	TINYINT(1) NOT NULL,
	IS_PROF			TINYINT(1) NOT NULL
)
ENGINE = MyISAM;
